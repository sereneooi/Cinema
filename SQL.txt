--Role
CREATE TABLE Role(
    roleID VARCHAR(5) NOT NULL PRIMARY KEY,
    role VARCHAR(15) NOT NULL UNIQUE
);

--Account
CREATE TABLE Account(
    emailAddress VARCHAR(50) NOT NULL PRIMARY KEY,
    name VARCHAR(50),
    password VARCHAR(100) NOT NULL,
    ic VARCHAR(14) NOT NULL UNIQUE,
    gender CHAR(1),
    birthDate DATE,
    address VARCHAR(100),
    contactNo VARCHAR(12),
    roleId VARCHAR(5) NOT NULL,
    dateCreated DATE,
    dateModified DATE,
    CONSTRAINT FK_role FOREIGN KEY (roleId) REFERENCES Role(roleId)
);

--Language
CREATE TABLE Language (
   languageID VARCHAR(5) NOT NULL,
   languageDesc VARCHAR(15),
   PRIMARY KEY(languageID)
);

--MovieStatus
CREATE TABLE MovieStatus (
   movieStatusID VARCHAR(5) NOT NULL,
   movieStatusDesc VARCHAR(15),
   PRIMARY KEY(movieStatusID)
);

--MovieType
CREATE TABLE MovieType (
   movieTypeID VARCHAR(5) NOT NULL,
   movieTypeDesc VARCHAR(15),
   PRIMARY KEY(movieTypeID)
);

--MovieRestriction
CREATE TABLE MovieRestriction (
   movieRestrictID VARCHAR(5) NOT NULL,
   restrictionDesc VARCHAR(10),
   restrictionImage VARCHAR(255),
   PRIMARY KEY(movieRestrictID)
);

--Movie
CREATE TABLE Movie (
   movieID VARCHAR(5) NOT NULL,
   movieName VARCHAR(50),
   movieTrailer VARCHAR(255),
   moviePoster VARCHAR(255),
   movieAdsPoster VARCHAR(255),
   synopsis VARCHAR(255),
   duration VARCHAR(30),
   availableDate DATE,
   movieStatusID VARCHAR(5) NOT NULL,
   movieTypeID VARCHAR(5) NOT NULL,
   movieRestrictID VARCHAR(5) NOT NULL,
   languageID VARCHAR(5) NOT NULL,
   PRIMARY KEY(movieID),
   FOREIGN KEY(movieStatusID) REFERENCES MovieStatus(movieStatusID),
   FOREIGN KEY(movieTypeID) REFERENCES MovieType(movieTypeID),
   FOREIGN KEY(movieRestrictID) REFERENCES MovieRestriction(movieRestrictID),
   FOREIGN KEY(languageID) REFERENCES Language(languageID)
);


--Actor
CREATE TABLE Actor (
   actorID VARCHAR(5) NOT NULL,
   actorName VARCHAR(50),
   PRIMARY KEY(actorID)
);

--CastList
CREATE TABLE CastList (
   castListID VARCHAR(5) NOT NULL,
   actorID VARCHAR(5) NOT NULL,
   movieID VARCHAR(5) NOT NULL,
   PRIMARY KEY(castListID),
   FOREIGN KEY(actorID) REFERENCES Actor(actorID),
   FOREIGN KEY(movieID) REFERENCES Movie(movieID)
);

--THEATRECLASSES
CREATE TABLE THEATRECLASSES(
    theatreClassesID varchar(6) NOT NULL,
    theatreClassesDesc varchar(50) NOT NULL,
    PRIMARY KEY (theatreClassesID)
);

--THEATRE
CREATE TABLE THEATRE(
    theatreID varchar(5) NOT NULL,
    theatreDesc varchar(50) NOT NULL,
    totalSeatCount int NOT NULL,
    theatreClassesID varchar(6) NOT NULL,
    PRIMARY KEY (theatreID),
    FOREIGN KEY (theatreClassesID) REFERENCES THEATRECLASSES(theatreClassesID)
);

--MovieSchedule
CREATE TABLE MovieSchedule (
   movieScheduleID VARCHAR(5) NOT NULL,
   showDate DATE,
   showTime TIME,
   showDateTime VARCHAR(255),
   movieID VARCHAR(5) NOT NULL,
   theatreID VARCHAR(5) NOT NULL,
   PRIMARY KEY(movieScheduleID),
   FOREIGN KEY(movieID) REFERENCES Movie(movieID),
   FOREIGN KEY(theatreID) REFERENCES THEATRE(theatreID)
);

--Promotion
CREATE TABLE Promotion (
        promoCode VARCHAR(5) NOT NULL,
        promoType VARCHAR(10) NOT NULL,
        promoAmount DOUBLE NOT NULL,
        promoStatus VARCHAR(20) NOT NULL,
        PRIMARY KEY (promoCode)
); 

--Payment
CREATE TABLE Payment (
        paymentID VARCHAR(5) NOT NULL,
        paymentDateTime TIMESTAMP NOT NULL,
        totalAmount DOUBLE NOT NULL,
        paymentMethod VARCHAR(30) NOT NULL,
        promoCode VARCHAR(5),
        PRIMARY KEY(paymentID),
        FOREIGN KEY (promoCode) REFERENCES Promotion(promoCode)  
);

--Booking
CREATE TABLE BOOKING(
    bookingID VARCHAR(5),
    bookingDateTime TIMESTAMP,
    emailAddress VARCHAR(50) NOT NULL,
    paymentID VARCHAR(5) NOT NULL,
    MODIFIEDDATETIME TIMESTAMP,
    PRIMARY KEY (bookingID),
    FOREIGN KEY (emailAddress) REFERENCES Account(emailAddress),
    FOREIGN KEY (paymentID) REFERENCES PAYMENT(paymentID)
);

--FnbCategory
CREATE TABLE FnbCategory (
        fnbCategoryID VARCHAR(6) NOT NULL,
        fnbCategory VARCHAR(50) NOT NULL UNIQUE,
        dateCreated VARCHAR(255) NOT NULL, 
        dateModified VARCHAR(255), 
        PRIMARY KEY(fnbCategoryID)
);

--Fnb
CREATE TABLE Fnb (
        fnbID VARCHAR(5) NOT NULL,
        fnbImage VARCHAR(255),
        fnbDescription VARCHAR(50) NOT NULL UNIQUE,
        fnbPrice FLOAT NOT NULL,
        fnbStockLeft INTEGER NOT NULL,
        active BOOLEAN NOT NULL,
        dateCreated VARCHAR(255) NOT NULL, 
        dateModified VARCHAR(255),
        fnbCategoryID VARCHAR(6) NOT NULL,
        PRIMARY KEY(fnbID),
        FOREIGN KEY(fnbCategoryID) REFERENCES FnbCategory(fnbCategoryID)
);

--FnbOrder
CREATE TABLE FnbOrder (
        fnbOrderID VARCHAR(6) NOT NULL,
        fnbOrderDateTime VARCHAR(255) NOT NULL,
        deliveryAddress VARCHAR(255),
        deliveryDateTime VARCHAR(255),
        deliveryStatus VARCHAR(30),
        bookingID VARCHAR(5) NOT NULL,
        PRIMARY KEY(fnbOrderID),
        FOREIGN KEY(bookingID) REFERENCES Booking(bookingID)    
);

--FnbOrderLine
CREATE TABLE FnbOrderLine (
        fnbOrderLineID VARCHAR(7) NOT NULL,
        quantity INTEGER NOT NULL,
        fnbID VARCHAR(5) NOT NULL,
        fnbOrderID VARCHAR(6) NOT NULL,
        PRIMARY KEY(fnbOrderLineID),
        FOREIGN KEY(fnbID) REFERENCES Fnb(fnbID),
        FOREIGN KEY(fnbOrderID) REFERENCES FnbOrder(fnbOrderID)
);

--SeatType
CREATE TABLE SEATTYPE(
    seatTypeID VARCHAR(6) NOT NULL,
    seatDescription VARCHAR(50),
    seatTypePrice DOUBLE,
    PRIMARY KEY (seatTypeID)
);

--Seat
CREATE TABLE SEAT(
    seatID VARCHAR(5) NOT NULL,
    SeatNo VARCHAR(5) NOT NULL,
    movieScheduleID VARCHAR(5) NOT NULL,
    seatTypeID VARCHAR(6) NOT NULL,
    PRIMARY KEY(seatID),
    FOREIGN KEY (movieScheduleID) REFERENCES MovieSchedule(movieScheduleID),
    FOREIGN KEY (seatTypeID) REFERENCES SEATTYPE(seatTypeID)
);


--TicketType
CREATE TABLE TICKETTYPE(
    ticketTypeID VARCHAR(6) NOT NULL,
    ticketTypeDesc VARCHAR(50) NOT NULL,
    ticketDiscountRate DOUBLE,
    PRIMARY KEY (ticketTypeID)
);

--Ticket
CREATE TABLE TICKET(
    ticketID VARCHAR(5) NOT NULL,
    seatID VARCHAR(5) NOT NULL,
    ticketTypeID VARCHAR(6) NOT NULL,
    BookingID VARCHAR(5) NOT NULL,
    PRIMARY KEY (ticketID),
    FOREIGN KEY (seatID) REFERENCES Seat(seatID),
    FOREIGN KEY (ticketTypeID) REFERENCES TICKETTYPE(ticketTypeID),
    FOREIGN KEY (bookingID) REFERENCES BOOKING(bookingID)
);

--Question
CREATE TABLE Question (
        questionID VARCHAR(5) NOT NULL,
        question VARCHAR(100) NOT NULL UNIQUE,
        active BOOLEAN NOT NULL,
        dateCreated VARCHAR(255) NOT NULL, 
        dateModified VARCHAR(255),
        PRIMARY KEY(questionID)
);

--FeedbackForm
CREATE TABLE FeedbackForm (
        feedbackFormID VARCHAR(6) NOT NULL, 
        feedbackDateTime VARCHAR(255) NOT NULL,
        bookingID VARCHAR(5) NOT NULL,
        PRIMARY KEY(feedbackFormID),
        FOREIGN KEY(bookingID) REFERENCES Booking(bookingID)
);

--FeedbackResponse
CREATE TABLE FeedbackResponse (
        feedbackResponseID VARCHAR(6) NOT NULL,
        questionID VARCHAR(5) NOT NULL,
        feedbackFormID VARCHAR(6) NOT NULL, 
        rating INTEGER,
        PRIMARY KEY(feedbackResponseID, questionID, feedbackFormID),
        FOREIGN KEY(questionID) REFERENCES Question(questionID),
        FOREIGN KEY(feedbackFormID) REFERENCES FeedbackForm(feedbackFormID)
);

--FavItem
CREATE TABLE FavItem (
        favItemID VARCHAR(6) NOT NULL,
        fnbID VARCHAR(5) NOT NULL,
        emailAddress VARCHAR(50) NOT NULL,
        PRIMARY KEY(favItemID),
        FOREIGN KEY(fnbID) REFERENCES Fnb(fnbID),
        FOREIGN KEY(emailAddress) REFERENCES Account(emailAddress)
);
